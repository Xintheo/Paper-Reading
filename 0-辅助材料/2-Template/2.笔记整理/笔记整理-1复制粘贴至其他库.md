// ========== 路径格式转换函数 ==========
function normalizePath(inputPath) {
    const path = require('path');
    let normalized = path.normalize(inputPath);
    return normalized;
}

// ========== 提取笔记中的所有 [[链接]] ==========
function extractWikiLinks(content) {
    const linkPattern = /\[\[([^\]|]+)(?:\|[^\]]+)?\]\]/g;
    const links = new Set();
    let match;
    
    while ((match = linkPattern.exec(content)) !== null) {
        links.add(match[1].trim());
    }
    
    return Array.from(links);
}

// ========== 递归收集所有链接的文件（不包括起始文件）==========
async function collectLinkedFiles(file, vault, processedFiles = new Set()) {
    const filePath = file.path;
    
    // 避免重复处理
    if (processedFiles.has(filePath)) {
        return [];
    }
    
    processedFiles.add(filePath);
    const linkedFiles = [];
    
    // 读取文件内容
    const content = await vault.read(file);
    
    // 提取所有链接
    const linkedFileNames = extractWikiLinks(content);
    
    // 递归处理每个链接的文件
    for (const linkedFileName of linkedFileNames) {
        // 尝试通过文件名搜索
        const files = vault.getMarkdownFiles();
        const matchedFile = files.find(f => f.basename === linkedFileName);
        
        if (matchedFile && !processedFiles.has(matchedFile.path)) {
            linkedFiles.push(matchedFile);
            // 递归收集该文件的链接
            const subFiles = await collectLinkedFiles(matchedFile, vault, processedFiles);
            linkedFiles.push(...subFiles);
        }
    }
    
    return linkedFiles;
}

// ========== 复制文件函数 ==========
async function copyFileToExternal(file, targetFolderPath, vault) {
    const fs = require('fs');
    const path = require('path');
    
    const content = await vault.read(file);
    const originalName = file.basename;
    
    // 生成不重复的文件名
    let newFileName = `${originalName}.md`;
    let targetFilePath = path.join(targetFolderPath, newFileName);
    let counter = 1;
    
    while (fs.existsSync(targetFilePath)) {
        newFileName = `${originalName} (${counter}).md`;
        targetFilePath = path.join(targetFolderPath, newFileName);
        counter++;
    }
    
    // 写入文件
    fs.writeFileSync(targetFilePath, content, 'utf8');
    
    return {
        originalFile: file,
        newFileName: newFileName,
        targetFilePath: targetFilePath
    };
}
