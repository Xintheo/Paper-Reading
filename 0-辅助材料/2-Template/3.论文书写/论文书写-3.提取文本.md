<%*
// 获取选中的文本
const selection = tp.file.selection().trim();
// 如果没有选中文本,提示用户
if (!selection) {
    new Notice('请先选中要处理的文本!', 2000);
    tR = "";
    return;
}
// 分割成行
const lines = selection.split('\n');
// 存储提取的结果
const results = [];
// 处理每一行
for (const line of lines) {
    // 跳过已完成的项目(带 [x] 的)
    if (line.includes('- [x]')) {
        continue;
    }
    
    // 匹配在 callout 内的二级列表:> > - [ ] 开头
    const calloutMatch = line.match(/^> > - \[ \] (.+)$/);
    
    if (calloutMatch) {
        // 提取内容(去除 > > - [ ] 后的文本)
        const content = calloutMatch[1].trim();
        results.push(content);
        continue;
    }
    
    // 兼容原格式:制表符或2-4个空格开头的 - [ ]
    const normalMatch = line.match(/^[\t ]{1,4}- \[ \] (.+)$/);
    
    if (normalMatch) {
        // 提取内容(去除 - [ ] 后的文本)
        const content = normalMatch[1].trim();
        results.push(content);
    }
}
// 检查是否找到内容
if (results.length === 0) {
    new Notice('未找到符合条件的第2级列表项!', 2000);
    tR = selection; // 返回原文本
    return;
}
// 合并成一个段落
const merged = results.join(' ');
new Notice(`✅ 已提取 ${results.length} 条内容并替换原文本!`, 2000);
// 返回提取的内容,替换选中的文本
tR = merged;
%>
