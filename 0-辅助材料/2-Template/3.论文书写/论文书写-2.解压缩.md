<%*
// 获取选中的文本
const selection = tp.file.selection();

if (!selection) {
    new Notice('请先选中要处理的文本！', 2000);
    return;
}

// 检测当前文本格式
const hasVersionCallout = selection.match(/^> \[!论文书写\]-/m);
const hasParagraphCallout = selection.match(/^> \[!段落书写\]-/m);

let options, selectedOption;

// 根据检测结果提供不同的选项
if (hasVersionCallout) {
    // 版本callout格式 → 只能解压为段落
    options = ["解压为段落"];
    selectedOption = await tp.system.suggester(options, options, false, "检测到版本格式，请选择操作:");
    
    if (!selectedOption) {
        new Notice('已取消操作', 2000);
        return;
    }
    
    // 执行：版本 → 段落
    const lines = selection.split('\n');
    const result = [];
    
    for (const line of lines) {
        // 跳过版本callout标题行
        if (line.match(/^> \[!论文书写\]-/)) {
            continue;
        }
        
        // 空的引用行 → 空行分隔
        if (line.trim() === '>') {
            result.push('');
            continue;
        }
        
        // 段落callout标题（> > [!段落书写]-）
        const paragraphMatch = line.match(/^> > (\[!段落书写\]-.+)$/);
        if (paragraphMatch) {
            result.push(`> ${paragraphMatch[1]}`);
            continue;
        }
        
        // summary callout（> > > [!summary]-）
        const summaryMatch = line.match(/^> > > (\[!summary\]-.+)$/);
        if (summaryMatch) {
            result.push(`> > ${summaryMatch[1]}`);
            continue;
        }
        
        // 内容行（> > > ...）
        const contentMatch = line.match(/^> > > (.+)$/);
        if (contentMatch) {
            result.push(`> > ${contentMatch[1]}`);
            continue;
        }
        
        // 空的二级引用（> >）
        if (line.trim() === '> >') {
            result.push('>');
            continue;
        }
        
        // 其他情况
        if (line.trim().startsWith('> >')) {
            const cleaned = line.replace(/^> > /, '> ');
            result.push(cleaned);
        }
    }
    
    const finalResult = result.join('\n');
    new Notice('✅ 已解压为独立段落！', 2000);
    tR = finalResult;
    
} else if (hasParagraphCallout) {
    // 段落callout格式 → 可以解压为草稿
    options = ["解压为草稿（原始列表）"];
    selectedOption = await tp.system.suggester(options, options, false, "检测到段落格式，请选择操作:");
    
    if (!selectedOption) {
        new Notice('已取消操作', 2000);
        return;
    }
    
    // 执行：段落 → 草稿
    const lines = selection.split('\n');
    const result = [];
    
    for (const line of lines) {
        // 跳过段落callout标题行（> [!段落书写]-）
        if (line.match(/^> \[!段落书写\]-/)) {
            continue;
        }
        
        // 检测summary callout开始（> > [!summary]- - [ ] ...）
        const summaryMatch = line.match(/^> > \[!summary\]- (- \[ \] .+)$/);
        if (summaryMatch) {
            result.push(summaryMatch[1]);
            continue;
        }
        
        // 检测二级任务列表（> > - [ ] ...）
        const level2Match = line.match(/^> > - \[ \] (.+)$/);
        if (level2Match) {
            result.push(`\t- [ ] ${level2Match[1]}`);
            continue;
        }
        
        // 检测其他二级内容（> > ...）
        const level2OtherMatch = line.match(/^> > (.+)$/);
        if (level2OtherMatch && !level2OtherMatch[1].startsWith('[!')) {
            result.push(`\t${level2OtherMatch[1]}`);
            continue;
        }
        
        // 跳过空引用行（>）
        if (line.trim() === '>') {
            continue;
        }
    }
    
    const finalResult = result.join('\n');
    new Notice('✅ 已转换回原始列表格式！', 2000);
    tR = finalResult;
    
} else {
    // 既不是版本也不是段落格式
    new Notice('⚠️ 选中的文本不是callout格式！', 2000);
    tR = selection;
}
%>
