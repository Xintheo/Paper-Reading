<%*
// 获取选中的文本
const selection = tp.file.selection();

// 检查是否有选中文本
if (!selection || !selection.trim()) {
    new Notice('❌ 请先选择要处理的文本!', 2000);
    tR = "";
    return;
}

// 创建选择菜单
const choice = await tp.system.suggester(
    ["1️⃣ 按句子切分 → 任务列表", "2️⃣ 中英文配对 → 嵌套任务"],
    ["split", "pair"]
);

// 如果用户取消选择
if (!choice) {
    new Notice('❌ 操作已取消', 2000);
    tR = "";
    return;
}

let formattedText = "";

// ======== 功能1: 按句子切分 ========
if (choice === "split") {

    // 第一步：将句号后紧跟的连续引用链接整体移到句号前
    // 处理：.[1](url),[2](url) → [1](url),[2](url).
    // 处理：.[1,2](url) → [1,2](url).
    // 处理：.[9](url) → [9](url).
    const textNormalized = selection.replace(
        /\.((?:\[\d+(?:,\d+)*\]\([^)]+\),?)+)/g,
        '$1.'
    );

    // 第二步：合并换行
    const textMerged = textNormalized.replace(/\n+/g, ' ');

    // 第三步：临时替换 Markdown 链接和裸URL，避免被误切
    const urlPlaceholders = [];
    const textWithoutUrls = textMerged.replace(
        /\[([^\]]*)\]\([^)]+\)|https?:\/\/[^\s]+/g,
        (match) => {
            urlPlaceholders.push(match);
            return `__URL${urlPlaceholders.length - 1}__`;
        }
    );

    // 第四步：切分句子
    const sentences = textWithoutUrls
        .split(/([。!！?？]|\.(?=\s|$))\s*/)
        .reduce((acc, curr, index, array) => {
            if (index % 2 === 0 && curr.trim()) {
                const punctuation = array[index + 1] || '';
                acc.push(curr.trim() + punctuation);
            }
            return acc;
        }, [])
        .filter(sentence => sentence.trim().length > 0)
        // 第五步：还原链接占位符
        .map(sentence => sentence.replace(/__URL(\d+)__/g, (_, i) => urlPlaceholders[i]));

    // 为每个句子添加任务复选框
    formattedText = sentences
        .map(sentence => `- [ ] ${sentence.trim()}`)
        .join('\n');

    new Notice(`✅ 已转换 ${sentences.length} 个句子!`, 2000);
}

// ======== 功能2: 中英文配对 ========
else if (choice === "pair") {
    // 按行分割文本
    const lines = selection.split('\n').map(line => line.trim()).filter(line => line);

    // 分离中文句子和英文句子
    const chineseSentences = [];
    const englishSentences = [];

    for (const line of lines) {
        // 移除已有的 - [ ] 标记
        const cleanLine = line.replace(/^-\s*\[\s*\]\s*/, '').trim();

        if (!cleanLine) continue;

        // 判断是否为中文句子（包含中文字符）
        if (/[\u4e00-\u9fa5]/.test(cleanLine)) {
            chineseSentences.push(cleanLine);
        } else {
            englishSentences.push(cleanLine);
        }
    }

    // 检查数量是否匹配
    if (chineseSentences.length !== englishSentences.length) {
        new Notice(`⚠️ 中文句子(${chineseSentences.length})和英文句子(${englishSentences.length})数量不匹配!`, 3000);
    }

    // 配对并格式化
    const pairedSentences = [];
    const maxLength = Math.max(chineseSentences.length, englishSentences.length);

    for (let i = 0; i < maxLength; i++) {
        if (i < chineseSentences.length) {
            pairedSentences.push(`- [ ] ${chineseSentences[i]}`);

            if (i < englishSentences.length) {
                pairedSentences.push(`    - [ ] ${englishSentences[i]}`);
            }
        } else if (i < englishSentences.length) {
            pairedSentences.push(`- [ ] ${englishSentences[i]}`);
        }
    }

    formattedText = pairedSentences.join('\n');

    new Notice(`✅ 已配对 ${Math.min(chineseSentences.length, englishSentences.length)} 组句子!`, 2000);
}

// 替换选中的文本
tR = formattedText;
%>
