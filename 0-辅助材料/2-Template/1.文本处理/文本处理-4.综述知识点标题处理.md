<%*
// 综述正文标题降级：选中内容后运行
// 1. 在选中内容最前面插入 ## 知识点
// 2. 所有标题降一级：## → ###，### → ####，#### → #####

const editor = app.workspace.activeLeaf.view.editor;
const selection = editor.getSelection();

if (!selection) {
    new Notice("请先选中需要处理的综述正文内容");
    return;
}

// 标题降级：从高级往低级替换，避免连锁替换
// 先处理 ##### → ######，再 #### → #####，依次到 ## → ###
let result = selection;
result = result.replace(/^######\s/gm, "####### ");
result = result.replace(/^#####\s/gm, "###### ");
result = result.replace(/^####\s/gm, "##### ");
result = result.replace(/^###\s/gm, "#### ");
result = result.replace(/^##\s/gm, "### ");

// 在最前面加 ## 知识点
result = "## 知识点\n\n" + result;

editor.replaceSelection(result);
new Notice("✓ 已添加 ## 知识点 并降级所有标题");
%>
