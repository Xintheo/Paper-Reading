<%*
const selection = tp.file.selection().trim();

// 交互选择功能
const options = ["📝 标注", "🔬 方法原理", "Figure图例"];
const choice = await tp.system.suggester(options, options, true, "请选择格式化功能：");

if (!choice) {
    new Notice("已取消操作", 2000);
    tR = selection;
    return;
}

let result = "";

if (choice === options[0]) {
    // --- 功能1: Summary ---
    const lines = selection.split('\n');
    const title = lines[0];
    const content = lines.slice(1).map(line => `> ${line}`).join('\n');
    result = `> [!summary]- ${title}\n${content}`;

} else if (choice === options[1]) {
    // --- 功能2: 方法原理 ---
    const content = selection.split('\n').map(line => `> ${line}`).join('\n');
    result = `> [!方法原理]- \n${content}`;

} else if (choice === options[2]) {
    // --- 功能3: 图片格式转换 ---
    const imageRegex = /!\[(.*?)\]\((.*?)\)/;
    const match = selection.match(imageRegex);
    if (match) {
        const altText = match[1];
        const imageUrl = match[2];
        result = `> [!figure]- ${altText}\n> ![${altText}](${imageUrl})`;
    } else {
        new Notice("未检测到有效的图片 Markdown 格式！", 3000);
        tR = selection;
        return;
    }
}

// 统一复制到剪贴板
navigator.clipboard.writeText(result).then(() => {
    new Notice(`✅ 已复制到剪贴板！`, 2000);
}).catch(err => {
    new Notice("复制失败: " + err.message, 3000);
});

tR = result;
%>
