<%*
// 获取选中的文本
const selection = tp.file.selection();

// 如果没有选中文本，提示用户
if (!selection) {
    new Notice('请先选择要处理的文本！', 3000);
    return;
}

// 创建选择菜单
const choice = await tp.system.suggester(
    ["1-AI处理", "2-笔记插入"],
    ["ai", "note"]
);

// 如果用户取消选择
if (!choice) {
    new Notice('操作已取消', 2000);
    tR = selection; // 返回原文本
    return;
}

// 处理选择
if (choice === "ai") {
    // ========== 选择1：AI处理 ==========
    const aiPrompt = `以下是一篇笔记的知识点或者几个不同的mermaid表格：

${selection}

---
你是一位擅长知识结构可视化的生物医学专家，请根据我提供的笔记内容，从生物医学逻辑角度出发，结合流程、病理机制、实验设计或治疗决策树的特点，分析内容之间的结构性和逻辑性关系，并使用合适的Mermaid图类型将其可视化，具体要求如下：
#### 1. 图类型选择与设计建议：
1. 请根据笔记内容特点，优先考虑以下几种图类型中最适合表达逻辑结构的一种（只需要一种即可；首选flowchart）：
    - 🦴 鱼骨图：用于分析病因、机制、诊断路径等因果关系； 
    - 🧭 思维导图：用于整理复杂知识点、分支概念或诊断分级； 
    - 🔁 流程图：适用于诊疗路径、实验流程、临床处理决策；
    - 🕸️ 网络图：用于表达多因素交互、通路整合、共存机制； 
2. 如果笔记信息结构层次较复杂，可以考虑以下方式使得图像简洁清晰：
	1. 可考虑多个子图或分模块结构
	2. 如果每个因素显示过小，因果关系可以使用→，或者↑，↓等表示，这样可以合并为一个因素，从而化简图像
#### 2. 图层结构要求：
- 每一层级信息不宜超过 3～5 个节点，必要时可进一步拆分为下一级；
- 保持结构递进，避免在一层级中堆砌大量概念；
- 尽量保证结构立体，避免单一线程逻辑，同一层级知识点可进行归纳
- 每个节点命名尽量精炼（不超过10个字），必要时可附注释（使用 \`:::note\` 或注释标签）。
#### 3. 输出要求：
- 只输出 **Mermaid代码**，无需图像描述；
- 图像开头请明确图类型（如 \`graph TD\`、\`graph LR\`、\`flowchart TB\`、\`fishbone\`等）；
- 若结构复杂，允许拆为多个图，按模块分别输出；
- 确保不遗漏任何一个笔记中的重要知识点或逻辑关系；
- 如果某些知识点之间关系不清晰，请合理假设其潜在关系，并在注释中标明（例如 \`%% 假设为因果关系\`）。`;

    // 复制到剪贴板
    await navigator.clipboard.writeText(aiPrompt);
    new Notice('AI指令已复制到剪贴板！可以直接粘贴给AI处理', 3000);
    
    // 🔑 关键：返回原文本，保持不变
    tR = selection;

} else if (choice === "note") {
    // ========== 选择2：笔记插入 ==========
    const result = `> [!mermaid]- \n> \`\`\`mermaid\n> ${selection.replace(/\n/g, '\n> ')}\n> \`\`\``;
    
    // 复制到剪贴板
    await navigator.clipboard.writeText(result);
    new Notice('Mermaid折叠代码块已复制到剪贴板并已插入笔记！', 2000);
    
    // 🔑 关键：返回格式化后的内容，替换原文本
    tR = result;
}
%>
