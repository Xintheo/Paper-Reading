<%*
const targetDirection = await tp.system.suggester(
    ["转换为 LR (左到右)", "转换为 TB (上到下)"],
    ["LR", "TB"],
    false,
    "请选择要转换的方向"
);
if (!targetDirection) {
    new Notice("已取消操作");
    return;
}
let content = await tp.file.content;

// 同时匹配 graph 和 flowchart，捕获关键字本身
const regex = /(flowchart|graph)(?:\s+(TB|TD|LR|RL|BT))?(?=\s|$)/g;
let matchCount = 0;
const newContent = content.replace(regex, (match, keyword) => {
    matchCount++;
    return `${keyword} ${targetDirection}`;  // 保留原关键字，只换方向
});

if (matchCount > 0) {
    const file = app.workspace.getActiveFile();
    await app.vault.modify(file, newContent);
    new Notice(`已转换 ${matchCount} 个流程图方向为 ${targetDirection}`);
} else {
    new Notice("没有找到 graph 或 flowchart 定义");
}
-%>
