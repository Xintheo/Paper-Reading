<%*
// 提供选项菜单
const options = ["添加内部链接", "添加网络链接", "修改文献格式"];
const selectedOption = await tp.system.suggester(options, options, false, "请选择操作:");

if (!selectedOption) {
    new Notice('已取消操作', 2000);
    tR = "";
    return;
}

// ========== 选项1：添加内部链接 ============
if (selectedOption === "添加内部链接") {
    const result = `[[]]`;
    new Notice('✅ 已添加内部链接！', 2000);
    tR = result;
}

// ========== 选项2：添加网络链接 ============
else if (selectedOption === "添加网络链接") {
    const result = `[]()`;
    new Notice('✅ 已添加网络链接格式！', 2000);
    tR = result;
}

// ========== 选项3：修改文献格式 ============
else if (selectedOption === "修改文献格式") {
    // 获取选中的文本
    const selection = tp.file.selection();

    if (!selection) {
        new Notice('请先选中要处理的文本！', 2000);
        tR = "";
        return;
    }
    
    const trimmedSelection = selection.trim();
    
    // 检测是否为有效的 URL
    const urlRegex = /^https?:\/\/.+/;
    
    if (urlRegex.test(trimmedSelection)) {
        // 生成格式化后的结果
        const result = `[PDF](${trimmedSelection})`;
        
        // 自动复制到剪贴板
        await navigator.clipboard.writeText(result);
        
        new Notice('✅ 链接已转换为PDF格式并复制到剪贴板！', 2000);
        
        tR = result;
    } else {
        new Notice('⚠️ 未检测到有效的URL格式！', 3000);
        tR = trimmedSelection; // 如果不是URL，返回原始选择
    }
}
%>
