
<%*
// 文稿整理模板选择
const templateChoices = [
    { 
        name: "📝 解压缩", 
        file: "0-辅助材料/2-Template/3.论文书写/论文书写-2.解压缩" 
    },
    { 
        name: "📝 提取文本", 
        file: "0-辅助材料/2-Template/3.论文书写/论文书写-3.提取文本" 
    }
];

// 显示选择菜单
const choice = await tp.system.suggester(
    templateChoices.map(c => c.name),
    templateChoices
);

if (!choice) {
    tR = "";
} else {
    try {
        // 使用tp.file.include加载模板
        const templateContent = await tp.file.include(`[[${choice.file}]]`);
        tR = templateContent;
    } catch (error) {
        try {
            // 备用方法:直接读取文件
            const templateFile = app.vault.getAbstractFileByPath(choice.file + ".md");
            if (templateFile) {
                const content = await app.vault.read(templateFile);
                tR = content;
            } else {
                new Notice(`找不到模板文件: ${choice.file}.md`);
                tR = "";
            }
        } catch (error2) {
            new Notice(`加载模板失败: ${error2.message}`);
            tR = "";
        }
    }
}
%>
