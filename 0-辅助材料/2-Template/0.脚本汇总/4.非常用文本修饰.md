<%*  
// Obsidian 文本格式化快捷脚本 - 其他格式  
// 使用方法: 选中文本后运行此模板(或不选中使用"取消格式"清理全文)

// 获取当前选中的文本  
const selectedText = tp.file.selection();

// 定义其他格式化选项  
const formatOptions = [  
"下划线 (Underline)",  
"上标 (Superscript)",  
"下标 (Subscript)",  
"删除线 (Strikethrough)",  
"行内代码 (Inline Code)",  
"取消格式 (Remove Format)"  
];

// 如果没有选中文本  
if (!selectedText) {  
const choice = await tp.system.suggester(formatOptions, formatOptions);

// 只有"取消格式"可以在未选中文本时使用  
if (choice === "取消格式 (Remove Format)") {  
    // 获取全文内容  
    const fullContent = await tp.file.content;  
      
    // 移除全文中所有格式标记  
    const cleanedContent = fullContent  
        .replace(/\*\*\*(.+?)\*\*\*/g, '$1')     // 加粗+斜体  
        .replace(/\*\*(.+?)\*\*/g, '$1')         // 加粗  
        .replace(/\*(.+?)\*/g, '$1')             // 斜体  
        .replace(/==(.+?)==/g, '$1')             // 高亮  
        .replace(/<u>(.+?)<\/u>/g, '$1')         // 下划线  
        .replace(/<sup>(.+?)<\/sup>/g, '$1')     // 上标  
        .replace(/<sub>(.+?)<\/sub>/g, '$1')     // 下标  
        .replace(/~~(.+?)~~/g, '$1')             // 删除线  
        .replace(/`(.+?)`/g, '$1');              // 行内代码  
      
    // 清空当前内容并替换为清理后的内容  
    const editor = app.workspace.activeLeaf.view.editor;  
    editor.setValue(cleanedContent);  
      
    new Notice("已移除全文格式标记!");  
    return;  
} else {  
    new Notice("请先选中要格式化的文本!");  
    return;  
}  
}

// 显示选择菜单  
const choice = await tp.system.suggester(formatOptions, formatOptions);

// 根据选择应用不同的格式  
let formattedText;

switch(choice) {  
case "下划线 (Underline)":  
formattedText = `<u>${selectedText}</u>`;  
break;  
case "上标 (Superscript)":  
formattedText = `<sup>${selectedText}</sup>`;  
break;  
case "下标 (Subscript)":  
formattedText = `<sub>${selectedText}</sub>`;  
break;  
case "删除线 (Strikethrough)":  
formattedText = `~~${selectedText}~~`;  
break;  
case "行内代码 (Inline Code)":  
formattedText = `\`${selectedText}\``;  
break;  
case "取消格式 (Remove Format)":  
// 移除选中文本中所有格式标记  
formattedText = selectedText  
    .replace(/\*\*\*(.+?)\*\*\*/g, '$1')     // 加粗+斜体  
    .replace(/\*\*(.+?)\*\*/g, '$1')         // 加粗  
    .replace(/\*(.+?)\*/g, '$1')             // 斜体  
    .replace(/==(.+?)==/g, '$1')             // 高亮  
    .replace(/<u>(.+?)<\/u>/g, '$1')         // 下划线  
    .replace(/<sup>(.+?)<\/sup>/g, '$1')     // 上标  
    .replace(/<sub>(.+?)<\/sub>/g, '$1')     // 下标  
    .replace(/~~(.+?)~~/g, '$1')             // 删除线  
    .replace(/`(.+?)`/g, '$1');              // 行内代码  
break;  
default:  
formattedText = selectedText;  
}

// 保存本次选择的格式  
try {  
await app.vault.adapter.write(  
".templater-last-format.json",  
JSON.stringify({format: choice})  
);  
} catch (e) {  
// 忽略保存错误  
}

// 输出格式化后的文本(会替换选中的文本)  
tR = formattedText;  
%>
