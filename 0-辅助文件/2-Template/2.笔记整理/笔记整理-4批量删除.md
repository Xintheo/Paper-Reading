<%*
// ========== 批量删除表格中列出的笔记（自动识别"名称"列）==========
// ⚠️ 危险操作：请确认后再执行

// ========== 从当前笔记内容中提取笔记名称 ==========
function extractNoteNames(content) {
const lines = content.split('\n');
const noteNames = [];
let colIdx = -1;
let state = 0;
for (const line of lines) {
    const t = line.trim();
    if (!(t.startsWith('|') && t.endsWith('|'))) {
        state = 0; colIdx = -1; continue;
    }
    const cells = t.slice(1, -1).split('|').map(c => c.trim());
    if (state === 0) {
        for (let i = 0; i < cells.length; i++) {
            if (cells[i] === '名称') { colIdx = i; state = 1; break; }
        }
        continue;
    }
    if (state === 1) {
        if (cells.every(c => /^[\s\-:]*$/.test(c))) { state = 2; }
        else { state = 0; colIdx = -1; }
        continue;
    }
    if (state === 2) {
        if (colIdx >= 0 && colIdx < cells.length) {
            let n = cells[colIdx].trim();
            if (!n) continue;
            const m = n.match(/\[\[([^\]|]+)/);
            if (m) n = m[1].trim();
            if (n) noteNames.push(n);
        }
    }
}
return [...new Set(noteNames)];
}

// ========== 主流程 ==========
const activeFile = app.workspace.getActiveFile();
if (!activeFile) {
new Notice("❌ 没有打开的笔记");
return;
}

try {
// ========== 第一步：提取笔记名称 ==========
new Notice("🔍 正在解析当前笔记中的笔记名称...");
const currentContent = await app.vault.read(activeFile);
const noteNames = extractNoteNames(currentContent);

if (noteNames.length === 0) {
    new Notice("❌ 未在当前笔记中找到有效的笔记名称");
    return;
}

new Notice(`📋 找到 ${noteNames.length} 个笔记名称`);

// ========== 第二步：匹配 Vault 中的笔记文件 ==========
const allFiles = app.vault.getMarkdownFiles();
const matchedFiles = [];
const unmatchedNames = [];

for (const name of noteNames) {
    const matchedFile = allFiles.find(f => f.basename === name);
    if (matchedFile) {
        matchedFiles.push({
            file: matchedFile,
            noteName: name
        });
    } else {
        unmatchedNames.push(name);
    }
}

if (unmatchedNames.length > 0) {
    new Notice(`⚠️ 有 ${unmatchedNames.length} 个笔记未找到匹配文件`, 4000);
    console.log("未匹配的笔记名称：");
    unmatchedNames.forEach((name, i) => console.log(`  ${i + 1}. ${name}`));
}

if (matchedFiles.length === 0) {
    new Notice("❌ 没有找到任何匹配的笔记文件");
    return;
}

// ========== 第三步：直接执行永久删除 ==========
console.log(`=== 待删除笔记列表（共 ${matchedFiles.length} 个）===`);
matchedFiles.forEach((entry, index) => {
    console.log(`  ${index + 1}. ${entry.noteName} → ${entry.file.path}`);
});

let deleteCount = 0;
let errorCount = 0;
const errorFiles = [];

new Notice(`🗑️ 正在永久删除 ${matchedFiles.length} 个笔记...`);

for (let i = 0; i < matchedFiles.length; i++) {
    const entry = matchedFiles[i];
  
    try {
        const fileExists = app.vault.getAbstractFileByPath(entry.file.path);
        if (!fileExists) {
            console.warn(`文件已不存在，跳过：${entry.noteName}`);
            continue;
        }
  
        await app.vault.delete(entry.file);
        deleteCount++;
  
        if ((i + 1) % 10 === 0) {
            new Notice(`⏳ 删除进度：${i + 1}/${matchedFiles.length}`, 1500);
        }
    } catch (err) {
        errorCount++;
        errorFiles.push(entry.noteName);
        console.error(`删除 "${entry.noteName}" 失败：`, err);
    }
}

// ========== 第四步：操作总结 ==========
let summaryMsg = `=== 删除操作总结 ===\n`;
summaryMsg += `删除方式：永久删除\n`;
summaryMsg += `✅ 成功删除：${deleteCount} 个\n`;

if (errorCount > 0) {
    summaryMsg += `❌ 删除失败：${errorCount} 个\n`;
    summaryMsg += `失败文件：${errorFiles.join(', ')}\n`;
}

if (unmatchedNames.length > 0) {
    summaryMsg += `⚠️ 未匹配（未处理）：${unmatchedNames.length} 个\n`;
}

console.log(summaryMsg);
new Notice(`🗑️ 已永久删除 ${deleteCount} 个笔记${errorCount > 0 ? `，${errorCount} 个失败` : ''}`, 5000);

} catch (error) {
new Notice(`❌ 操作失败：${error.message}`, 6000);
console.error("批量删除出错：", error);
console.error("错误堆栈：", error.stack);
}
%>
