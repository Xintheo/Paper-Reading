<%*
// 获取选中的文本
const selection = tp.file.selection();

if (!selection) {
    new Notice('请先选中要处理的文本！', 2000);
    tR = "";
    return;
}

// 第一步：选择类型（段落或版本）
const types = ["段落", "版本"];
const selectedType = await tp.system.suggester(types, types, false, "请选择类型:");

if (!selectedType) {
    new Notice('已取消操作', 2000);
    tR = "";
    return;
}

// ========== 辅助函数：统计一级任务数量 ==========
function countLevel1Tasks(text) {
    const lines = text.split('\n');
    let count = 0;
    for (const line of lines) {
        if (line.match(/^- \[ \] /)) {
            count++;
        }
    }
    return count;
}

// ========== 辅助函数：统计段落数量 ==========
function countParagraphs(text) {
    const paragraphMatches = text.match(/> \[!段落书写\]-/g);
    return paragraphMatches ? paragraphMatches.length : 0;
}

// ========== 辅助函数：提取逻辑链 ==========
function extractLogicChain(text) {
    const lines = text.split('\n');
    for (const line of lines) {
        const trimmed = line.trim();
        if (trimmed && !trimmed.match(/^- \[ \] /)) {
            return trimmed;
        }
    }
    return '';
}

// ========== 从段落标题中提取知识线（括号和冒号之间的内容） ==========
function extractKnowledgeLineFromParagraph(paragraphHeader) {
    // 匹配格式：> [!段落书写]- 第N段：X句（知识线：详细内容）
    // 提取 （ 和 ： 之间的内容
    const match = paragraphHeader.match(/（([^：]+)：/);
    if (match) {
        return match[1].trim();
    }
    return '';
}

// ========== 提取所有段落的知识线并组合成新的逻辑链 ==========
function extractAndCombineKnowledgeLines(text) {
    const lines = text.split('\n');
    const knowledgeLines = [];
    
    for (const line of lines) {
        // 匹配段落书写的标题行（选中文本中的段落标题）
        if (line.match(/^> \[!段落书写\]-/)) {
            const knowledgeLine = extractKnowledgeLineFromParagraph(line);
            if (knowledgeLine) {
                knowledgeLines.push(knowledgeLine);
            }
        }
    }
    
    // 用 → 连接所有知识线
    return knowledgeLines.join('→');
}

// ========== 分支1：段落转换 ============
if (selectedType === "段落") {
    const options = ["第1段", "第2段", "第3段", "第4段", "第5段"];
    const selected = await tp.system.suggester(options, options, false, "请选择段落:");
    
    if (!selected) {
        new Notice('已取消操作', 2000);
        tR = "";
        return;
    }
    
    const logicChain = extractLogicChain(selection);
    const sentenceCount = countLevel1Tasks(selection);
    const lines = selection.split('\n');
    const result = [];
    
    if (logicChain) {
        result.push(`> [!段落书写]- ${selected}：${sentenceCount}句（${logicChain}）`);
    } else {
        result.push(`> [!段落书写]- ${selected}：${sentenceCount}句`);
    }
    
    let isInGroup = false;
    let skipFirstNonTask = false;
    
    for (const line of lines) {
        const trimmed = line.trim();
        
        if (trimmed === '') {
            continue;
        }
        
        if (!skipFirstNonTask && !trimmed.match(/^- \[ \] /)) {
            skipFirstNonTask = true;
            continue;
        }
        
        const level1Match = line.match(/^- \[ \] (.+)$/);
        if (level1Match) {
            if (isInGroup) {
                result.push('>');
            }
            result.push(`> > [!summary]- ${line}`);
            isInGroup = true;
            continue;
        }
        
        const level2Match = line.match(/^[\t ]{1,4}- \[ \] (.+)$/);
        if (level2Match) {
            result.push(`> > - [ ] ${level2Match[1]}`);
            continue;
        }
        
        const trimmedLine = line.replace(/^[\t ]+/, '');
        if (trimmedLine) {
            result.push(`> > ${trimmedLine}`);
        }
    }
    
    const finalResult = result.join('\n');
    new Notice(`✅ 已生成段落书写 callout（${sentenceCount}句）！`, 2000);
    tR = finalResult;
}

// ========== 分支2：版本转换 ============
else if (selectedType === "版本") {
    const options = ["第1版本", "第2版本", "第3版本", "第4版本", "第5版本"];
    const selected = await tp.system.suggester(options, options, false, "请选择版本:");
    
    if (!selected) {
        new Notice('已取消操作', 2000);
        tR = "";
        return;
    }
    
    const hasParagraphCallout = selection.match(/> \[!段落书写\]-/m);
    
    if (hasParagraphCallout) {
        const paragraphCount = countParagraphs(selection);
        const combinedKnowledgeLines = extractAndCombineKnowledgeLines(selection);
        const lines = selection.split('\n');
        const result = [];
        
        // 格式：第N版本（X段）：知识线1→知识线2
        if (combinedKnowledgeLines) {
            result.push(`> [!论文书写]- ${selected}（${paragraphCount}段）：${combinedKnowledgeLines}`);
        } else {
            result.push(`> [!论文书写]- ${selected}（${paragraphCount}段）`);
        }
        
        for (const line of lines) {
            if (line.trim() === '') {
                result.push('>');
                continue;
            }
            result.push(`> ${line}`);
        }
        
        const finalResult = result.join('\n');
        new Notice(`✅ 已将段落包装为版本 callout（${paragraphCount}段）！`, 2000);
        tR = finalResult;
        
    } else {
        const sentenceCount = countLevel1Tasks(selection);
        const lines = selection.split('\n');
        const result = [];
        result.push(`> [!论文书写]- ${selected}：${sentenceCount}句`);
        
        let isInGroup = false;
        
        for (const line of lines) {
            if (line.trim() === '') {
                continue;
            }
            
            const level1Match = line.match(/^- \[ \] (.+)$/);
            if (level1Match) {
                if (isInGroup) {
                    result.push('>');
                }
                result.push(`> > [!summary]- ${line}`);
                isInGroup = true;
                continue;
            }
            
            const level2Match = line.match(/^[\t ]{1,4}- \[ \] (.+)$/);
            if (level2Match) {
                result.push(`> > - [ ] ${level2Match[1]}`);
                continue;
            }
            
            const trimmedLine = line.replace(/^[\t ]+/, '');
            if (trimmedLine) {
                result.push(`> > ${trimmedLine}`);
            }
        }
        
        const finalResult = result.join('\n');
        new Notice(`✅ 已生成论文书写 callout（${sentenceCount}句）！`, 2000);
        tR = finalResult;
    }
}
