
<%*
// Obsidian 文本格式化快捷脚本 - 常用格式
// 使用方法: 选中文本后运行此模板

// 获取当前选中的文本
const selectedText = tp.file.selection();

// 如果没有选中文本,提示用户
if (!selectedText) {
    new Notice("请先选中要格式化的文本!");
    return;
}

// 定义常用格式化选项
const formatOptions = [
    "加粗 (Bold)",
    "斜体 (Italic)",
    "高亮 (Highlight)",
    "加粗+斜体 (Bold+Italic)"
];

// 显示选择菜单
const choice = await tp.system.suggester(formatOptions, formatOptions);

// 根据选择应用不同的格式
let formattedText;

switch(choice) {
    case "加粗 (Bold)":
        formattedText = `**${selectedText}**`;
        break;
    case "斜体 (Italic)":
        formattedText = `*${selectedText}*`;
        break;
    case "高亮 (Highlight)":
        formattedText = `==${selectedText}==`;
        break;
    case "加粗+斜体 (Bold+Italic)":
        formattedText = `***${selectedText}***`;
        break;
    default:
        formattedText = selectedText;
}

// 保存本次选择的格式
try {
    await app.vault.adapter.write(
        ".templater-last-format.json", 
        JSON.stringify({format: choice})
    );
} catch (e) {
    // 忽略保存错误
}

// 输出格式化后的文本(会替换选中的文本)
tR = formattedText;
%>
