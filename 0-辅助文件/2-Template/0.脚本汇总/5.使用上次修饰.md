<%*  
// 使用上次的格式化选项（支持常规和非常规格式）  
// 使用方法: 选中文本后运行此模板

// 获取当前选中的文本  
const selectedText = tp.file.selection();

if (!selectedText) {  
new Notice("请先选中要格式化的文本!");  
return;  
}

// 读取上次保存的格式  
let lastFormat = null;  
try {  
const content = await app.vault.adapter.read(".templater-last-format.json");  
lastFormat = JSON.parse(content).format;  
} catch (e) {  
new Notice("没有历史格式记录! 请先使用一次格式化功能。");  
return;  
}

// 应用上次的格式（支持常规 + 非常规）  
let formattedText;

switch(lastFormat) {  
// 常规格式  
case "加粗 (Bold)":  
formattedText = `**${selectedText}**`;  
break;  
case "斜体 (Italic)":  
formattedText = `*${selectedText}*`;  
break;  
case "高亮 (Highlight)":  
formattedText = `==${selectedText}==`;  
break;  
case "加粗+斜体 (Bold+Italic)":  
formattedText = `***${selectedText}***`;  
break;

// 非常规格式  
case "下划线 (Underline)":  
    formattedText = `<u>${selectedText}</u>`;  
    break;  
case "上标 (Superscript)":  
    formattedText = `<sup>${selectedText}</sup>`;  
    break;  
case "下标 (Subscript)":  
    formattedText = `<sub>${selectedText}</sub>`;  
    break;  
case "删除线 (Strikethrough)":  
    formattedText = `~~${selectedText}~~`;  
    break;  
case "行内代码 (Inline Code)":  
    formattedText = `\`${selectedText}\``;  
    break;  
case "取消格式 (Remove Format)":  
    formattedText = selectedText  
        .replace(/\*\*\*(.+?)\*\*\*/g, '$1')     // 加粗+斜体  
        .replace(/\*\*(.+?)\*\*/g, '$1')         // 加粗  
        .replace(/\*(.+?)\*/g, '$1')             // 斜体  
        .replace(/==(.+?)==/g, '$1')             // 高亮  
        .replace(/<u>(.+?)<\/u>/g, '$1')         // 下划线  
        .replace(/<sup>(.+?)<\/sup>/g, '$1')     // 上标  
        .replace(/<sub>(.+?)<\/sub>/g, '$1')     // 下标  
        .replace(/~~(.+?)~~/g, '$1')             // 删除线  
        .replace(/`(.+?)`/g, '$1');              // 行内代码  
    break;  
  
default:  
    formattedText = selectedText;  
}

// 输出格式化后的文本  
tR = formattedText;  
%>
