<%*
// 定义模板选择项
const templateChoices = [
    { name: "📝 Mermaid", file: "0-辅助材料/2-Template/1.文本处理/文本处理-1.Mermaid" },
    { name: "🔬 Mermaid格式修改", file: "0-辅助材料/2-Template/1.文本处理/文本处理-2.Mermaid格式修改" },
    { name: "📊 折叠", file: "0-辅助材料/2-Template/1.文本处理/文本处理-3.折叠" },
    { name: "📑 综述知识点标题处理", file: "0-辅助材料/2-Template/1.文本处理/文本处理-4.综述知识点标题处理" }
];

// 显示选择菜单
const choice = await tp.system.suggester(
    templateChoices.map(c => c.name),
    templateChoices
);

if (!choice) {
    tR = "";
} else {
    try {
        // 方法1：使用tp.file.include（去掉.md扩展名）
        const templateContent = await tp.file.include(`[[${choice.file}]]`);
        tR = templateContent;
    } catch (error) {
        try {
            // 方法2：直接读取文件
            const templateFile = app.vault.getAbstractFileByPath(choice.file + ".md");
            if (templateFile) {
                const content = await app.vault.read(templateFile);
                tR = content;
            } else {
                new Notice(`找不到模板文件: ${choice.file}.md`);
                tR = "";
            }
        } catch (error2) {
            new Notice(`加载模板失败: ${error2.message}`);
            tR = "";
        }
    }
}
%>
